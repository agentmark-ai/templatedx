# Main

<ForEach arr={props.messageHistory}>
  {(item) => (
    <>
      <If condition={item.role == 'user'}>
        user: {item.message}
      </If>
      <ElseIf condition={item.role == 'assistant'}>
        assistant: {item.message}
      </ElseIf>
    </>
  )}
</ForEach>
